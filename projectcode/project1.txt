void timee(FILE *file) {
    // Get the current time
    time_t currentTime;
    struct tm *localTime;

    time(&currentTime);
    localTime = localtime(&currentTime);

    // Display the current time
    fprintf(file, "Current time: %02d:%02d:%02d\n",
            localTime->tm_hour, localTime->tm_min, localTime->tm_sec);
}
